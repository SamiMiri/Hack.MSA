let state = null;

function freshState(startMoney = 0) {
  return {
    money: startMoney,
    stress: 0,
    knowledge: 0,
    score: 0,
    turn: 0,
    flags: new Set(),
    pendingEvents: [],
    history: [],
    currentScenario: null,
    sceneMap: null, // reference to current scenario's scenes
  };
}

function scheduleEvent(turnsFromNow, sceneId) {
  state.pendingEvents.push({ triggerTurn: state.turn + turnsFromNow, sceneId });
}

function checkPendingEvents() {
  const due = state.pendingEvents.find(e => e.triggerTurn <= state.turn);
  if (due) {
    state.pendingEvents = state.pendingEvents.filter(e => e !== due);
    return due.sceneId;
  }
  return null;
}

function renderScene(sceneId, pendingFeedback) {
  const scene = state.sceneMap[sceneId];
  if (!scene) { renderEnding("ending_mid"); return; }
  if (scene.ending) { renderEnding(sceneId); return; }

  const game = document.getElementById("game");
  const fb = pendingFeedback
    ? `<div class="feedback ${pendingFeedback.kind === 'good' ? 'good' : pendingFeedback.kind === 'bad' ? 'bad' : 'warn'}">${pendingFeedback.text}</div>`
    : "";

  game.innerHTML = `
    ${fb}
    <div class="scene">
      <div class="title">${scene.title}</div>
      ${scene.text(state)}
    </div>
    <div class="choices" id="choices"></div>
  `;

  const letters = "ABCDEFGH";
  const choicesEl = document.getElementById("choices");
  // Shuffle choice order so "good" isn't always A
  const order = scene.choices.map((_, i) => i);
  for (let i = order.length - 1; i > 0; i--) {
    const j = Math.floor(Math.random() * (i + 1));
    [order[i], order[j]] = [order[j], order[i]];
  }
  order.forEach((origIdx, displayIdx) => {
    const c = scene.choices[origIdx];
    const btn = document.createElement("button");
    btn.className = "choice";
    btn.innerHTML = `<span class="letter">[${letters[displayIdx]}]</span>${c.label}`;
    btn.onclick = () => pickChoice(sceneId, origIdx);
    choicesEl.appendChild(btn);
  });
  window.scrollTo({ top: 0, behavior: "smooth" });
}

function pickChoice(sceneId, choiceIdx) {
  const scene = state.sceneMap[sceneId];
  const choice = scene.choices[choiceIdx];
  choice.effects(state);
  state.turn += 1;
  state.history.push({ sceneId, choice: choice.label });
  renderStats();

  if (state.stress >= 100) { renderEnding("ending_death"); return; }

  let nextId = choice.next ? choice.next(state) : choice.nextId;

  const pending = checkPendingEvents();
  if (pending && nextId !== null && state.sceneMap[pending]) {
    const resumeId = nextId;
    nextId = pending;
    state.sceneMap[pending]._resumeTo = resumeId;
  }

  const feedback = choice.feedback ? { kind: choice.kind, text: choice.feedback } : null;

  if (nextId === null) {
    const resume = scene._resumeTo || "finale";
    renderScene(resume, feedback);
  } else {
    renderScene(nextId, feedback);
  }
}

function renderEnding(endingId) {
  const e = state.sceneMap[endingId] || {
    kind: "meh", title: "END", text: () => "The story ends here."
  };
  const game = document.getElementById("game");
  game.innerHTML = `
    <div class="scene ending ${e.kind === 'good' ? 'good' : e.kind === 'bad' ? 'bad' : ''}">
      <h2>${e.title}</h2>
      <div style="margin: 14px 0;">${e.text(state).replace(/\n/g, '<br>')}</div>
      <div style="color: var(--muted); margin-top: 14px;">
        Final score: <b>${state.score}</b> · Knowledge: <b>${state.knowledge}</b> · Cash: <b>$${Math.round(state.money).toLocaleString()}</b>
      </div>
      <button class="restart" onclick="returnToMenu()">← Play another scenario</button>
      <button class="menu-btn ghost" onclick="startScenario('${state.currentScenario}')">Replay this one</button>
    </div>
  `;
  renderStats();
}

function renderMenu() {
  document.getElementById("top-bar").style.display = "none";
  document.getElementById("stats-wrap").style.display = "none";
  const game = document.getElementById("game");
  game.innerHTML = `
    <div class="scene" style="background: transparent; padding: 0 4px 12px;">
      <div class="title">Pick a scenario</div>
      Three separate stories. Each is about 10–14 decisions. Memory carries within a scenario — flee the IRS and they come back for you — but resets when you pick a new one.
    </div>
    ${Object.values(scenarios).map(sc => `
      <div class="menu-card" onclick="startScenario('${sc.id}')">
        <h3>${sc.name}</h3>
        <div class="who">${sc.who}</div>
        <div class="desc">${sc.desc}</div>
      </div>

  `;

}

function startScenario(id) {
  const sc = scenarios[id];
  state = freshState(sc.startMoney);
  state.currentScenario = id;
  state.sceneMap = sc.scenes;

  document.getElementById("top-bar").style.display = "flex";
  document.getElementById("stats-wrap").style.display = "block";
  document.getElementById("scenario-name").textContent = sc.name + " · " + sc.who;

  renderStats();
  renderScene(sc.startId, null);
}

function returnToMenu() {
  renderMenu();
}
